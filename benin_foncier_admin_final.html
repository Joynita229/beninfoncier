Instructions de déploiement
Copiez-collez ces scripts dans Supabase → SQL Editor → New Query. Exécutez dans l'ordre indiqué. Commencez par ⓪ (table profiles) si elle n'existe pas encore.
⓪ Table profiles — Rôles utilisateurs REQUIS EN PREMIER
Crée la table profiles liée à auth.users et ajoute la colonne role. Nécessaire pour le verrouillage par rôle de l'interface admin.
-- 1. Créer la table profiles (si elle n'existe pas) CREATE TABLE IF NOT EXISTS profiles ( id uuid REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY, name text, role text NOT NULL DEFAULT 'user', -- 'user' | 'vendor' | 'admin' created_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); -- 2. Activer RLS ALTER TABLE profiles ENABLE ROW LEVEL SECURITY; -- 3. Chaque utilisateur peut lire son propre profil CREATE POLICY "User read own profile" ON profiles FOR SELECT USING (auth.uid() = id); -- 4. Les admins ont accès complet CREATE POLICY "Admin full access on profiles" ON profiles USING ( EXISTS ( SELECT 1 FROM profiles p2 WHERE p2.id = auth.uid() AND p2.role = 'admin' ) ); -- 5. Créer automatiquement un profil à l'inscription CREATE OR REPLACE FUNCTION handle_new_user() RETURNS trigger AS $$ BEGIN INSERT INTO profiles (id, name, role) VALUES (NEW.id, NEW.raw_user_meta_data->>'name', 'user') ON CONFLICT (id) DO NOTHING; RETURN NEW; END; $$ LANGUAGE plpgsql SECURITY DEFINER; DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users; CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE PROCEDURE handle_new_user(); -- 6. Promouvoir un utilisateur en admin (remplacez l'UUID) -- UPDATE profiles SET role = 'admin' WHERE id = 'UUID_DE_VOTRE_COMPTE_ADMIN';
① Table bf_vendors — Gestion des vendeurs
Synchronisée à l'inscription OTP de chaque vendeur
CREATE TABLE IF NOT EXISTS bf_vendors ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, phone text NOT NULL UNIQUE, name text, email text, plan text DEFAULT 'free', plan_expires_at timestamptz, kyc_status text DEFAULT 'pending', -- pending | verified | rejected status text DEFAULT 'active', -- active | suspended affiliate_code text, created_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); ALTER TABLE bf_vendors ENABLE ROW LEVEL SECURITY; CREATE POLICY "Admin full access" ON bf_vendors USING (true);
② Table bf_affiliates — Programme de parrainage
Codes affiliés, clics, conversions et commissions
CREATE TABLE IF NOT EXISTS bf_affiliates ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, code text NOT NULL UNIQUE, phone text NOT NULL, name text, clicks int DEFAULT 0, conversions int DEFAULT 0, commission_pending int DEFAULT 0, commission_paid int DEFAULT 0, kyc_status text DEFAULT 'pending', status text DEFAULT 'active', created_at timestamptz DEFAULT now() ); ALTER TABLE bf_affiliates ENABLE ROW LEVEL SECURITY; CREATE POLICY "Admin full access" ON bf_affiliates USING (true);
③ Table bf_rif_requests — Dossiers RIF
Rapports d'inspection foncière soumis via la plateforme
CREATE TABLE IF NOT EXISTS bf_rif_requests ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, ref text NOT NULL, type text, -- standard | express phone text NOT NULL, zone text, parcelle text, coordgps text, montant int DEFAULT 0, status text DEFAULT 'pending', -- pending | processing | done rapport_url text, notes text, created_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); ALTER TABLE bf_rif_requests ENABLE ROW LEVEL SECURITY; CREATE POLICY "Admin full access" ON bf_rif_requests USING (true);
④ Table bf_blog — Articles du blog
Gestion des articles publiés sur la plateforme
CREATE TABLE IF NOT EXISTS bf_blog ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, titre text NOT NULL, categorie text, auteur text DEFAULT 'SIPI-AFRIK', extrait text, contenu text, image_url text, slug text UNIQUE, status text DEFAULT 'draft', -- draft | published published_at timestamptz, created_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); ALTER TABLE bf_blog ENABLE ROW LEVEL SECURITY; CREATE POLICY "Admin full access" ON bf_blog USING (true); CREATE POLICY "Public read" ON bf_blog FOR SELECT USING (status = 'published');
⑤ Table bf_admin_log — Journal des actions admin
Traçabilité des actions sensibles (RGPD / APDP)
CREATE TABLE IF NOT EXISTS bf_admin_log ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, action text NOT NULL, target_id text, target_type text, details jsonb, ip text, created_at timestamptz DEFAULT now() ); ALTER TABLE bf_admin_log ENABLE ROW LEVEL SECURITY; CREATE POLICY "Admin full access" ON bf_admin_log USING (true);
